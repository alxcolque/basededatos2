Sintaxis

Una función al igual que los stored procedures, su código se guarda en la base de datos y tiene como función realizar un cálculo y devolver un valor.

Ejemplos

  • Ejemplo 1.
  • Crear una función que sume dos números. 
    CREATE FUNCTION SUMAR_AYB(a number, b number)
return number
IS
C number;
BEGIN
C:=A + B;
return C;
END;

SELECT SUMAR_AYB(3,44)FROM DUAL;
  • Ejemplo 2.
  • Contar todos los empleados del esquema HR 
    CREATE OR REPLACE FUNCTION GET_TOTAL_EMPLEADOS
RETURN NUMBER
IS
E_TOTAL NUMBER:=0;
BEGIN
SELECT COUNT(*) INTO E_TOTAL FROM EMPLOYEES;
RETURN E_TOTAL;
END;

-- Probando la función:

SELECT GET_TOTAL_EMPLEADOS FROM DUAL;
  • Ejemplo 3.
  • Realizar un programa para sumar el salario mínimo y el salario máximo de los empleados de HR. 
    CREATE OR REPLACE FUNCTION SUMA_SALARIO
RETURN NUMBER
IS
SAL_TOT NUMBER;
MIN_SAL NUMBER;
MAX_SAL NUMBER;
BEGIN
SELECT MIN(SALARY)INTO MIN_SAL FROM EMPLOYEES;
SELECT MAX(SALARY)INTO MAX_SAL FROM EMPLOYEES;
SAL_TOT := MIN_SAL + MAX_SAL;
RETURN SAL_TOT;
END;

SELECT SUMA_SALARIO FROM DUAL;
  • Ejemplo 4.
  • Crear una función como parámetro una fecha, y que devuelva solo el año 
    CREATE OR REPLACE FUNCTION DEVOLVER_FECHA(FECHA DATE)
RETURN NUMBER
AS
V_YEAR NUMBER(4);
BEGIN
V_YEAR := EXTRACT(YEAR FROM FECHA);
--V_YEAR := TO_NUMBER(TO_CHAR(FECHA,'YYYY'));
RETURN V_YEAR;
END;

SELECT DEVOLVER_FECHA(SYSDATE) FROM DUAL;
  • Ejemplo 5.
  • Crear una función que reciba el ID de un empleado y que regrese el número de puestos que ha tenido dicho empleado. 
    CREATE OR REPLACE FUNCTION N_PUESTOS(empid NUMBER)
RETURN NUMBER IS
v_cont NUMBER(2);
BEGIN
SELECT COUNT(*) INTO v_cont
FROM job_history
WHERE employee_id = empid;
RETURN v_cont;
END;

BEGIN
DBMS_OUTPUT.PUT_LINE(N_PUESTOS(101));
END;
  • Ejemplo 6.
  • Crear una función que reciba el manager id y regrese el nombre de los empleados a cargo de dicho manager, los nombres deben ser retornados como una cadena separada por comas. 
    CREATE OR REPLACE FUNCTION MANAGER_A_CARGO(ID NUMBER)
RETURN VARCHAR2 AS
CURSOR C_EMP IS
SELECT FIRST_NAME FROM EMPLOYEES WHERE MANAGER_ID = ID;
REGISTRO C_EMP%ROWTYPE;
CAD VARCHAR2(255) := '';
BEGIN
FOR REGISTRO IN C_EMP LOOP
CAD:=CAD||REGISTRO.FIRST_NAME||', ';
END LOOP;
RETURN CAD;
END;

BEGIN
DBMS_OUTPUT.PUT_LINE(MANAGER_A_CARGO(101));
END;
